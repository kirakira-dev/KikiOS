Filesystem

void *open(const char *path)

Open a file or directory. Returns NULL on failure.

void close(void *file)

Close an open file.

int read(void *file, char *buf, size_t size, size_t offset)

Read size bytes at offset. Returns bytes read.

int write(void *file, const char *buf, size_t size)

Write size bytes (appends). Returns bytes written.

int is_dir(void *node)

Returns 1 if node is a directory.

int file_size(void *node)

Get file size in bytes.

void *create(const char *path)

Create a new file. Returns file handle.

void *mkdir(const char *path)

Create a directory.

int delete(const char *path)

Delete a file.

int rename(const char *path, const char *newname)

Rename a file or directory.

int readdir(void *dir, int index, char *name, size_t size, uint8_t *type)

Read directory entry at index. Returns 0 on success.

int set_cwd(const char *path), int get_cwd(char *buf, size_t size)

Set/get current working directory.
